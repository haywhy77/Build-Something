version: '3'
services:
  gateway-service:
    build: ./services/gateway
    ports:
      - "8000:8000"
    environment:
      - JWT_SECRET=${SECRET_KEY}
      - PYTHONUNBUFFERED=1
      - PYTHONPATH=shared
      - AUTH_SERVICE_HOST_URL=http://auth-service:8000/api/v1
      - RECIPE_SERVICE_HOST_URL=http://recipe-service:8000/api/v1
      - INGREDIENT_SERVICE_HOST_URL=http://ingredient-service:8000/api/v1
      - MEALPLAN_SERVICE_HOST_URL=http://mealplan-service:8000/api/v1
      - RATING_SERVICE_HOST_URL=http://rating-service:8000/api/v1
      - RABBITMQ_HOST=rabbitmq
    depends_on:
      db:
        condition: service_healthy
      rabbitmq:
        condition: service_healthy
    volumes:
      - ./services/gateway/:/app/
      - ./shared:/app/shared
      - ./.env:/app/.env
    networks:
      - recipe-network

  auth-service:
    build: ./services/auth
    ports:
      - "8001:8000"
    environment:
      - DATABASE_URL=mysql+pymysql://user:password@db/$recipe_user

      - PYTHONUNBUFFERED=1

      - recipe-network

  db:
    image: mysql:${MYSQL_VERSION:-8.0}
    command: --default-authentication-plugin=caching_sha2_password
    restart: ${DOCKER_RESTART_POLICY:-unless-stopped}
    environment:
      MYSQL_ROOT_PASSWORD: rootpassword
      MYSQL_DATABASE: recipe_db
      MYSQL_USER: user
      MYSQL_PASSWORD: password
    ports:
      - "${MYSQL_PORT:-3306}:3306"
    volumes:
      - ./shared/mysqldb:/var/lib/mysql
    networks:
      - recipe-network
    healthcheck:
      test: ["CMD", "mysqladmin", "ping", "-h", "localhost", "-u$$MYSQL_USER", "-p$$MYSQL_PASSWORD"]
      interval: 10s
      timeout: 5s
      retries: 5

  rabbitmq:
    image: rabbitmq:3.13-management
    container_name: rabbitmq
    ports:
      - "5672:5672"    # AMQP protocol port
      - "15672:15672"  # Management UI port
    healthcheck:
      test: ["CMD", "rabbitmq-diagnostics", "ping"]
      interval: 10s
      timeout: 5s
      retries: 5
    networks:
      - recipe-network
    restart: unless-stopped


volumes:
  mysql_data:
    name: recipe_management_mysql_data

networks:
  recipe-network:
    driver: bridge
